12hardware_interface::CallbackReturn LittlebotHardwareComponent::on_init(
13 const hardware_interface::HardwareInfo & info)
14{
15 if (hardware_interface::SystemInterface::on_init(info) !=
16 hardware_interface::CallbackReturn::SUCCESS)
17 {
18 return hardware_interface::CallbackReturn::FAILURE;
19 }
20
21 hw_positions_.resize(info_.joints.size(), std::numeric_limits<double>::quiet_NaN());
22 hw_velocities_.resize(info_.joints.size(), std::numeric_limits<double>::quiet_NaN());
23 hw_commands_.resize(info_.joints.size(), std::numeric_limits<double>::quiet_NaN());
24
25 for (const hardware_interface::ComponentInfo & joint : info_.joints)
26 {
27 if (joint.command_interfaces.size() != kNumCommandInterface_)
28 {
29 RCLCPP_FATAL(
30 rclcpp::get_logger("LittlebotSystemHardware"),
31 "Joint '%s' has %zu command interfaces found. 1 expected.", joint.name.c_str(),
32 joint.command_interfaces.size());
33 return hardware_interface::CallbackReturn::ERROR;
34 }
35
36 if (joint.command_interfaces[0].name != hardware_interface::HW_IF_VELOCITY)
37 {
38 RCLCPP_FATAL(
39 rclcpp::get_logger("LittlebotSystemHardware"),
40 "Joint '%s' have %s command interfaces found. '%s' expected.", joint.name.c_str(),
41 joint.command_interfaces[0].name.c_str(), hardware_interface::HW_IF_VELOCITY);
42 return hardware_interface::CallbackReturn::ERROR;
43 }
44
45 if (joint.state_interfaces.size() != kNumStateInterface_)
46 {
47 RCLCPP_FATAL(
48 rclcpp::get_logger("LittlebotSystemHardware"),
49 "Joint '%s' has %zu state interface. 2 expected.", joint.name.c_str(),
50 joint.state_interfaces.size());
51 return hardware_interface::CallbackReturn::ERROR;
52 }
53
54 if (joint.state_interfaces[0].name != hardware_interface::HW_IF_POSITION)
55 {
56 RCLCPP_FATAL(
57 rclcpp::get_logger("LittlebotSystemHardware"),
58 "Joint '%s' have %s state interfaces found. '%s' expected.", joint.name.c_str(),
59 joint.state_interfaces[0].name.c_str(), hardware_interface::HW_IF_POSITION);
60 return hardware_interface::CallbackReturn::ERROR;
61 }
62
63 if (joint.state_interfaces[1].name != hardware_interface::HW_IF_VELOCITY)
64 {
65 RCLCPP_FATAL(
66 rclcpp::get_logger("LittlebotSystemHardware"),
67 "Joint '%s' have %s state interfaces found. '%s' expected.", joint.name.c_str(),
68 joint.state_interfaces[1].name.c_str(), hardware_interface::HW_IF_VELOCITY);
69 return hardware_interface::CallbackReturn::ERROR;
70 }
71 }
72
73 return hardware_interface::CallbackReturn::SUCCESS;
74}

76hardware_interface::CallbackReturn LittlebotHardwareComponent::on_activate(
77 const rclcpp_lifecycle::State & /*previous_state*/)
78{
79 for (auto i = 0u; i < hw_positions_.size(); i++)
80 {
81 if (std::isnan(hw_positions_[i]))
82 {
83 hw_positions_[i] = 0;
84 hw_velocities_[i] = 0;
85 hw_commands_[i] = 0;
86 }
87 }
88
89 RCLCPP_INFO(rclcpp::get_logger("LittlebotSystemHardware"), "Successfully activated!");
90
91 return hardware_interface::CallbackReturn::SUCCESS;
92}

102std::vector<hardware_interface::StateInterface> LittlebotHardwareComponent::export_state_interfaces()
103{
104 std::vector<hardware_interface::StateInterface> state_interfaces;
105 for (auto i = 0u; i < info_.joints.size(); i++)
106 {
107 state_interfaces.emplace_back(hardware_interface::StateInterface(
108 info_.joints[i].name, hardware_interface::HW_IF_POSITION, &hw_positions_[i]));
109 state_interfaces.emplace_back(hardware_interface::StateInterface(
110 info_.joints[i].name, hardware_interface::HW_IF_VELOCITY, &hw_velocities_[i]));
111 }
112
113 return state_interfaces;
114}
